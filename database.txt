CREATE DATABASE IF NOT EXISTS projeto_db;
USE projeto_db;

CREATE TABLE utilizadores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    data_registo TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE atores_diretores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(150) NOT NULL,
    tipo ENUM('ator','diretor') NOT NULL
);

CREATE TABLE generos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE filmes_series (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(200) NOT NULL,
    sinopse TEXT,
    duracao INT, -- minutos
    diretor_id INT,
    ano_lancamento YEAR,
    poster_url VARCHAR(255),
    trailer_url VARCHAR(255),
    FOREIGN KEY (diretor_id) REFERENCES atores_diretores(id)
);

CREATE TABLE genero_filme (
    movie_id INT,
    genre_id INT,
    PRIMARY KEY (movie_id, genre_id),
    FOREIGN KEY (movie_id) REFERENCES filmes_series(id) ON DELETE CASCADE,
    FOREIGN KEY (genre_id) REFERENCES generos(id) ON DELETE CASCADE
);

CREATE TABLE movie_cast (
    movie_id INT,
    actor_id INT,
    PRIMARY KEY (movie_id, actor_id),
    FOREIGN KEY (movie_id) REFERENCES filmes_series(id) ON DELETE CASCADE,
    FOREIGN KEY (actor_id) REFERENCES atores_diretores(id) ON DELETE CASCADE
);


CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    movie_id INT NOT NULL,
    data_review TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    classificacao INT CHECK (classificacao BETWEEN 1 AND 10),
    critica TEXT,
    votos_utilidade INT DEFAULT 0,
    FOREIGN KEY (user_id) REFERENCES utilizadores(id) ON DELETE CASCADE,
    FOREIGN KEY (movie_id) REFERENCES filmes_series(id) ON DELETE CASCADE
);

CREATE TABLE review_votes (
    user_id INT,
    review_id INT,
    PRIMARY KEY (user_id, review_id),
    FOREIGN KEY (user_id) REFERENCES utilizadores(id) ON DELETE CASCADE,
    FOREIGN KEY (review_id) REFERENCES reviews(id) ON DELETE CASCADE
);


CREATE TABLE favoritos (
    user_id INT,
    movie_id INT,
    PRIMARY KEY (user_id, movie_id),
    FOREIGN KEY (user_id) REFERENCES utilizadores(id) ON DELETE CASCADE,
    FOREIGN KEY (movie_id) REFERENCES filmes_series(id) ON DELETE CASCADE
);

CREATE TABLE lista_utilizadores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    nome VARCHAR(150) NOT NULL,
    descricao TEXT,
    FOREIGN KEY (user_id) REFERENCES utilizadores(id) ON DELETE CASCADE
);

CREATE TABLE lista_filmes (
    list_id INT,
    movie_id INT,
    PRIMARY KEY (list_id, movie_id),
    FOREIGN KEY (list_id) REFERENCES lista_utilizadores(id) ON DELETE CASCADE,
    FOREIGN KEY (movie_id) REFERENCES filmes_series(id) ON DELETE CASCADE
);
